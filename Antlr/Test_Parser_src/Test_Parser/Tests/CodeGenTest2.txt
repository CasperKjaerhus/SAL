function main() returns void begin
	for(thisPin: 0...pinCount) begin
		digitalWrite(ledPins[thisPin], HIGH);
		delay(timer);
		digitalWrite(ledPins[thisPin], LOW);
	end
	
	for(thisPin: pinCount-1...0) begin
		digitalWrite(ledPins[thisPin], HIGH);
		delay(timer);
		digitalWrite(ledPins[thisPin], LOW);
	end
end

	for(thisPin: 0...pinCount) begin
